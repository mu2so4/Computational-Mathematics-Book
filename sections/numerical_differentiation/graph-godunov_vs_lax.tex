\documentclass[14pt]{extarticle}

\begin{document}
\begin{figure}[ht]
	\centering
	\newcommand*{\xlabel}{$x$}
	\newcommand*{\ylabel}{$v$}
	\newcommand*{\varA}{-2}
	\newcommand*{\varB}{3.5}
	\newcommand*{\varC}{0}
	\newcommand*{\Shift}{2}
	\newcommand*{\timeShift}{1}
	\newcommand*{\Height}{5cm}
	\newcommand*{\Width}{6.5cm}
	\pgfmathsetmacro{\yMin}{-0.4}
	\pgfmathsetmacro{\yMax}{\Shift + 0.4}
	\pgfmathsetmacro{\xMin}{\varA - 0.2}
	\pgfmathsetmacro{\xMax}{\varB + 0.2}

	\begin{tikzpicture}
		\begin{axis} [
			height=\Height,
			width=\Width,
			xlabel = {\xlabel},
			ylabel = {\ylabel},
			axis x line = middle,
			axis y line = middle,
			xmin = \xMin,
			xmax = \xMax,
			ymin = \yMin,
			ymax = \yMax,
			ticks = none ]

			\coordinate(A) at 	(\varA,		\timeShift);
			\coordinate(C1) at 	(\varC+\Shift,		\timeShift);
			\coordinate(C2) at 	(\varC+\Shift,		0);
			\coordinate(B) at 	(\varB,		0);

			\draw[line width=.08cm] (A) -- (C1) (C2) -- (B);

			\node[below] at 	(C2){$at$};

			\draw[dashed] (C1) -- (C2);
			\addplot[mark=*,only marks, fill=black] (\Shift, \timeShift)
				node[above, pos=1]{};
			\addplot[mark=*,only marks, fill=white] (\Shift, 0)
				node[above, pos=1]{};
		\end{axis}
	\end{tikzpicture}\qquad
	\begin{tikzpicture} [
		declare function = {
			f(\x) = (\x < 0) * \timeShift +
				(\x >= 0) * (\x <= \Shift * 1.3) * \timeShift *
					(1 - \x / \Shift / 1.3);
	},]
		\begin{axis} [
			height=\Height,
			width=\Width,
			xlabel = {\xlabel},
			ylabel = {\ylabel},
			axis x line = middle,
			axis y line = middle,
			xmin = \xMin,
			xmax = \xMax,
			ymin = \yMin,
			ymax = \yMax,
			ticks = none ]

			\pgfmathsetmacro{\newC}{\varC + \Shift}

			\coordinate(A) at 	(\varA,		\timeShift);
			\coordinate(C1) at 	(\varC+\Shift,		\timeShift);
			\coordinate(C2) at 	(\varC+\Shift,		0);
			\coordinate(B) at 	(\varB,		0);

			\node[below] at 	(C2){$at$};

			\draw[dashed] (C1) -- (C2) (A) -- (C1);
			\addplot[color = black, only marks, mark=*]{f(x)};
		\end{axis}
	\end{tikzpicture}\qquad
	\begin{tikzpicture} [
		declare function = {
			f(\x) = (\x < -\Shift * 3 / 4) * \timeShift +
				(\x >= -\Shift * 3 / 4) * (\x < -\Shift / 4) *
					\timeShift * 0.8 +
				(\x >= -\Shift / 4) * (\x < 0) *
					\timeShift * 0.6 +
				(\x >= 0) * (\x < \Shift / 2) *
					\timeShift * 0.4 +
				(\x >= \Shift / 2) * (\x <= \Shift) *
					\timeShift * 0.2;
	},]
		\begin{axis} [
			height=\Height,
			width=\Width,
			xlabel = {\xlabel},
			ylabel = {\ylabel},
			axis x line = middle,
			axis y line = middle,
			xmin = \xMin,
			xmax = \xMax,
			ymin = \yMin,
			ymax = \yMax,
			ticks = none ]

			\pgfmathsetmacro{\newC}{\varC + \Shift}

			\coordinate(A) at 	(\varA,		\timeShift);
			\coordinate(C1) at 	(\varC+\Shift,		\timeShift);
			\coordinate(C2) at 	(\varC+\Shift,		0);
			\coordinate(B) at 	(\varB,		0);

			\node[below] at 	(C2){$at$};

			\draw[dashed] (C1) -- (C2) (A) -- (C1);
			\addplot[color = black, only marks, mark=*]{f(x)};
		\end{axis}
	\end{tikzpicture}
\end{figure}
\end{document}
